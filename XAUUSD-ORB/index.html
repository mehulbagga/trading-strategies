//@version=5
strategy("XAUUSD Opening Range Breakout",
     overlay=true,
     default_qty_type=strategy.fixed,
     default_qty_value=1)

// ── Time references (New York EST) ──────────────────
ny1803 = timestamp("America/New_York", year, month,
          dayofmonth, 18, 3, 0)
ny1818 = timestamp("America/New_York", year, month,
          dayofmonth, 18, 18, 0)

isRangeBar   = (time >= ny1803 and time < ny1818)
isAfterRange = (time >= ny1818)

// ── Build opening range ──────────────────────────────
var float rangeHigh = na
var float rangeLow  = na

if isRangeBar
    rangeHigh := na(rangeHigh) ? high : math.max(rangeHigh, high)
    rangeLow  := na(rangeLow)  ? low  : math.min(rangeLow,  low)
